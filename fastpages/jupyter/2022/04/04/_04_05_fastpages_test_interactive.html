def getLinePolygonByTwoPoints(point1: list, point2: list, width: float) -> Polygon:
    if point1[0] == point2[0]:
        p1 = (point1[0] - width / 2, point1[1])
        p2 = (point1[0] + width / 2, point1[1])
        p3 = (point2[0] + width / 2, point2[1])
        p4 = (point2[0] - width / 2, point2[1])
    elif point1[1] == point2[1]:
        p1 = (point1[0], point1[1] - width / 2)
        p2 = (point1[0], point1[1] + width / 2)
        p3 = (point2[0], point2[1] + width / 2)
        p4 = (point2[0], point2[1] - width / 2)
    else:
        k = (point1[1] - point2[1]) / (point1[0] - point2[0])
        theta = math.atan(abs(k))
        if k < 0:
            p1 = (point1[0] - math.sin(theta) * width / 2, point1[1] - math.cos(theta) * width / 2)
            p2 = (point1[0] + math.sin(theta) * width / 2, point1[1] + math.cos(theta) * width / 2)
            p3 = (point2[0] + math.sin(theta) * width / 2, point2[1] + math.cos(theta) * width / 2)
            p4 = (point2[0] - math.sin(theta) * width / 2, point2[1] - math.cos(theta) * width / 2)
        if k > 0:
            p1 = (point1[0] + math.sin(theta) * width / 2, point1[1] - math.cos(theta) * width / 2)

            p4 = (point2[0] + math.sin(theta) * width / 2, point2[1] - math.cos(theta) * width / 2)

    line = Polygon([p1, p2, p3, p4])

    return line

def create_hexgrid_h(bbox, side):
    """
    returns an array of Points describing hexagons centers that are inside the given bounding_box
    :param bbox: The containing bounding box. x-y bounding box is a (minx, miny, maxx, maxy) tuple
    :param side: The size of the hexagons (edge length)
    :return: The hexagon grid
    """
    grid = []
    # vertical step
    v_step = math.sqrt(3) * side
    # horizontal step
    h_step = 1.5 * side

    x_min = min(bbox[0], bbox[2])
    x_max = max(bbox[0], bbox[2])
    y_min = min(bbox[1], bbox[3])
    y_max = max(bbox[1], bbox[3])

    h_skip = math.ceil(x_min / h_step) - 1 #roundup
    h_start = h_skip * h_step
    

    v_skip = math.ceil(y_min / v_step) - 1
    v_start = v_skip * v_step

    h_end = x_max + h_step
    v_end = y_max + v_step

    if v_start - (v_step / 2.0) < y_min:
        v_start_array = [v_start + (v_step / 2.0), v_start]
    else:
        v_start_array = [v_start - (v_step / 2.0), v_start]

    v_start_idx = int(abs(h_skip) % 2)
    
    c_x = h_start
    c_y = v_start_array[v_start_idx]
    v_start_idx = (v_start_idx + 1) % 2
    while c_x < h_end:
        while c_y < v_end:
            grid.append((c_x, c_y))
            c_y += v_step
        c_x += h_step
        c_y = v_start_array[v_start_idx]
        v_start_idx = (v_start_idx + 1) % 2
    #     print(v_start_idx)
    # print(v_skip, h_skip, v_start, h_start)
    return grid

def rotate_point(point, angle, center_point=(0, 0)):
    """Rotates a point around center_point(origin by default)
    Angle is in degrees.
    Rotation is counter-clockwise
    """
    angle_rad = math.radians(angle % 360)
    # Shift the point so that center_point becomes the origin
    new_point = (point[0] - center_point[0], point[1] - center_point[1])
    new_point = (new_point[0] * math.cos(angle_rad) - new_point[1] * math.sin(angle_rad),
                 new_point[0] * math.sin(angle_rad) + new_point[1] * math.cos(angle_rad))
    # Reverse the shifting we have done
    new_point = (new_point[0] + center_point[0], new_point[1] + center_point[1])
    return new_point
